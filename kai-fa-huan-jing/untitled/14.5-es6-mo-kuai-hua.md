# 14.5 ES6 模块化

{% code title="a.js" %}
```javascript
export function fn() {
    console.log('fn');
}

export const name = 'rdr';

export const obj = {
    a: '123'
}
```
{% endcode %}

{% code title="b.js" %}
```javascript
function fn() {
    console.log('fn');
}

const name = 'rdr';

const obj = {
    a: '123'
}

export {
    fn,
    name,
    obj
}
```
{% endcode %}

{% code title="c.js" %}
```javascript
const obj = {
    name: 'xxx'
}

const x = 1;

export default {
    obj,
    x
};
```
{% endcode %}

{% code title="index.js" %}
```javascript
import { fn, name, obj } from './a'; // 解构赋值

import { fn, name, obj } from './b'; // 解构赋值

import c from './c'; // c.obj c.x

fn();
console.log(name);
console.log(obj);
```
{% endcode %}
